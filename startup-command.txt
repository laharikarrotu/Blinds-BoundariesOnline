# Azure App Service Startup Commands
# Copy one of these commands to Azure Portal → Configuration → General settings → Startup Command

# Option 1: Use startup.py (recommended)
python startup.py

# Option 2: Use main.py (fallback)
python main.py

# Option 3: Direct uvicorn command
uvicorn main:application --host 0.0.0.0 --port 8000

# Option 4: Use startup script
bash startup.sh
